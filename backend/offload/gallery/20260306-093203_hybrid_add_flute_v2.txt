```python
import matplotlib.pyplot as plt
import numpy as np
from matplotlib.patches import Polygon, Circle, Ellipse, Patch, FancyArrowPatch
from matplotlib.collections import PatchCollection
from matplotlib.transforms import Affine2D

fig, ax = plt.subplots(figsize=(10, 6), facecolor='k')
plt.style.use('dark_background')

# Chordophone symbol
chordophone = Polygon([(0, 0), (0, 2), (1, 2), (1, 0)], fill=True, color='orange')
ax.add_patch(chordophone)

# Add holes
for i in range(1, 7):
    hole = Circle((i*0.2, 1.5), 0.15, fill=True, color='blue')
    ax.add_patch(hole)

# Add breath control
breath_control = Ellipse((0.5, 1.7), 0.4, 0.2, fill=True, color='orange')
ax.add_patch(breath_control)

# Add finger placement marks
finger_marks = Circle((0.2, 1.5), 0.15, fill=True, color='black')
ax.add_patch(finger_marks)

# Connect breath control to holes
for i in range(1, 7):
    arrow = FancyArrowPatch((0.5, 1.7), (i*0.2, 1.5), arrowstyle='<|-|>', mutation_scale=20, color='grey')
    ax.add_patch(arrow)

# Simulate spectrum
freqs = np.linspace(200, 2000, 100)
ampl = np.sin(freqs * np.pi / 100)
plt.plot(freqs, ampl, color='blue')

# Performer position
performer = Circle((0.1, 1.8), 0.05, fill=True, color='white', alpha=0.5)
ax.add_patch(performer)
ax.annotate('Performer', (0.1, 1.6), color='white')

# Spine settings
ax.spines['bottom'].set_color('black')
ax.spines['left'].set_color('black')
ax.spines['top'].set_visible(False)
ax.spines['right'].set_visible(False)
ax.spines['left'].set_alpha(0.5)

# Grid and ticks
ax.grid(True, color='grey', alpha=0.3)
ax.set_xticks([0, 1, 2, 3, 4, 5, 6])
ax.set_yticks([0, 1, 2])

# Labeling
ax.text(0.5, 2.1, 'Chordophone', ha='center', fontsize=14, color='white')

plt.show()
```
