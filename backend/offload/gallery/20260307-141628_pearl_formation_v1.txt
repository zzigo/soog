## organogram (matplotlib)
```python
import matplotlib.pyplot as plt
import numpy as np

fig, ax = plt.subplots(figsize=(10, 6))
ax.set_facecolor('k')
plt.style.use('dark_background')

# Central chamber (Pearl Formation) - rectangle with orange color
rect_center = plt.Rectangle((2, 3), 5, 2, edgecolor='white', facecolor='#FFA500', linewidth=2.5)
ax.add_patch(rect_center)

# Surrounding chambers - three beige rectangles
beige_color = '#D2B48C'
for i in range(3):
    rect_surround = plt.Rectangle((1.5 + 6 * i, 1), 4, 1, edgecolor='white', facecolor=beige_color, linewidth=2.5)
    ax.add_patch(rect_surround)

# Arrows indicating the flow of secretion
for i in range(3):
    start = (1.5 + 6 * i + 2, 0.8)
    end = (1.5 + 6 * i + 4 - 0.5, 0.8)
    ax.annotate('', xy=end, xytext=start,
                arrowprops=dict(facecolor='white', edgecolor='black', 
                                width=3, headwidth=7, headlength=7))

# Numbers representing nacre layers
ax.text(2.5, 4.1, '0.5 mm', fontsize=12, ha='center')

plt.axis('equal')
plt.title('Organogram of Pearl Formation Sound Generator')
plt.show()
```
