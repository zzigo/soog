meshes = []

    height=220,
    sections=64

# bell radiation (acoustic flare)
# ---------------------------------------------------

    radius=48,
    height=80,
    sections=64
)

bell.apply_translation((0, 0, -150))
meshes.append(bell)

# ---------------------------------------------------
# mouthpiece interface
# ---------------------------------------------------

mouthpiece = trimesh.creation.cylinder(
    radius=14,
    height=50,
    sections=64
)

mouthpiece.apply_translation((0, 0, 140))
meshes.append(mouthpiece)

# ---------------------------------------------------
# nacre layers (pearl-like acoustic shells)
# ---------------------------------------------------

layer_count = 5
layer_spacing = 8
layer_height = 28

for i in range(layer_count):

    r = 36 + i * layer_spacing

    layer = trimesh.creation.cylinder(
        radius=r,
        height=layer_height,
        sections=64
    )

    z = -20 + i * 30
    layer.apply_translation((0, 0, z))

    layer.visual.face_colors = [255, 230, 180, 180]

    meshes.append(layer)

# ---------------------------------------------------
# tension strings
# ---------------------------------------------------

string_count = 8
string_radius = 1
string_height = 210
string_circle = 42

angles = np.linspace(0, 2*np.pi, string_count, endpoint=False)

for a in angles:

    x = np.cos(a) * string_circle
    y = np.sin(a) * string_circle

    string = trimesh.creation.cylinder(
        radius=string_radius,
        height=string_height,
        sections=24
    )

    string.apply_translation((x, y, 0))
    meshes.append(string)

# ---------------------------------------------------
# top and bottom rings (string anchors)
# ---------------------------------------------------

top_ring = trimesh.creation.torus(
    major_radius=42,
    minor_radius=2
)

top_ring.apply_translation((0, 0, 105))
meshes.append(top_ring)

bottom_ring = trimesh.creation.torus(
    major_radius=42,
    minor_radius=2
)

bottom_ring.apply_translation((0, 0, -105))
meshes.append(bottom_ring)

# ---------------------------------------------------
# merge geometry
# ---------------------------------------------------

instrument = trimesh.util.concatenate(meshes)

# preview
instrument.show()

# ---------------------------------------------------
# optional export
# ---------------------------------------------------

instrument.export("pearl_sound_generator.glb")
